319 bool AABB::smits_intersect_ray(const Vector3 &from,const Vector3& dir, float t0, float t1) const {
320 
321  float divx=1.0/dir.x;
322  float divy=1.0/dir.y;
323  float divz=1.0/dir.z;
324 
325  Vector3 upbound=pos+size;
326  float tmin, tmax, tymin, tymax, tzmin, tzmax;
327  if (dir.x >= 0) {
328  tmin = (pos.x - from.x) * divx;
329  tmax = (upbound.x - from.x) * divx;
330  }
331  else {
332  tmin = (upbound.x - from.x) * divx;
333  tmax = (pos.x - from.x) * divx;
334  }
335  if (dir.y >= 0) {
336  tymin = (pos.y - from.y) * divy;
337  tymax = (upbound.y - from.y) * divy;
338  }
339  else {
340  tymin = (upbound.y - from.y) * divy;
341  tymax = (pos.y - from.y) * divy;
342  }
343  if ( (tmin > tymax) || (tymin > tmax) )
344  return false;
345  if (tymin > tmin)
346  tmin = tymin;
347  if (tymax < tmax)
348  tmax = tymax;
349  if (dir.z >= 0) {
350  tzmin = (pos.z - from.z) * divz;
351  tzmax = (upbound.z - from.z) * divz;
352  }
353  else {
354  tzmin = (upbound.z - from.z) * divz;
355  tzmax = (pos.z - from.z) * divz;
356  }
357  if ( (tmin > tzmax) || (tzmin > tmax) )
358  return false;
359  if (tzmin > tmin)
360  tmin = tzmin;
361  if (tzmax < tmax)
362  tmax = tzmax;
363  return ( (tmin < t1) && (tmax > t0) );
364 }
