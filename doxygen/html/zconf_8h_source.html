242 /* Maximum value for windowBits in deflateInit2 and inflateInit2.
243  * WARNING: reducing MAX_WBITS makes minigzip unable to extract .gz files
244  * created by gzip. (Files created by minigzip can still be extracted by
245  * gzip.)

248 # define MAX_WBITS 15 /* 32K LZ77 window */
249 #endif
250 
251 /* The memory requirements for deflate are (in bytes):
252  (1 << (windowBits+2)) + (1 << (memLevel+9))
253  that is: 128K for windowBits=15 + 128K for memLevel = 8 (default values)
254  plus a few kilobytes for small objects. For example, if you want to reduce
255  the default memory requirements from 256K to 128K, compile with
256  make CFLAGS="-O -DMAX_WBITS=14 -DMAX_MEM_LEVEL=7"
257  Of course this will generally degrade compression (there's no free lunch).
258 
259  The memory requirements for inflate are (in bytes) 1 << windowBits
260  that is, 32K for windowBits=15 (default value) plus a few kilobytes
261  for small objects.
262 */
