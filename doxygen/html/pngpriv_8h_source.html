1472 /* An internal API to validate the format of a floating point number.
1473  * The result is the index of the next character. If the number is
1474  * not valid it will be the index of a character in the supposed number.
1475  *
1476  * The format of a number is defined in the PNG extensions specification
1477  * and this API is strictly conformant to that spec, not anyone elses!
1478  *
1479  * The format as a regular expression is:
1480  *
1481  * [+-]?[0-9]+.?([Ee][+-]?[0-9]+)?
1482  *
1483  * or:
1484  *
1485  * [+-]?.[0-9]+(.[0-9]+)?([Ee][+-]?[0-9]+)?
1486  *
1487  * The complexity is that either integer or fraction must be present and the
1488  * fraction is permitted to have no digits only if the integer is present.
1489  *
1490  * NOTE: The dangling E problem.
1491  * There is a PNG valid floating point number in the following:
1492  *
1493  * PNG floating point numb1.ers are not greedy.
1494  *
1495  * Working this out requires *TWO* character lookahead (because of the
1496  * sign), the parser does not do this - it will fail at the 'r' - this
1497  * doesn't matter for PNG sCAL chunk values, but it requires more care
1498  * if the value were ever to be embedded in something more complex. Use
1499  * ANSI-C strtod if you need the lookahead.
1500  */
1501 /* State table for the parser. */
1502 #define PNG_FP_INTEGER 0 /* before or in integer */
1503 #define PNG_FP_FRACTION 1 /* before or in fraction */
1504 #define PNG_FP_EXPONENT 2 /* before or in exponent */
1505 #define PNG_FP_STATE 3 /* mask for the above */
1506 #define PNG_FP_SAW_SIGN 4 /* Saw +/- in current state */
1507 #define PNG_FP_SAW_DIGIT 8 /* Saw a digit in current state */
1508 #define PNG_FP_SAW_DOT 16 /* Saw a dot in current state */
1509 #define PNG_FP_SAW_E 32 /* Saw an E (or e) in current state */
1510 #define PNG_FP_SAW_ANY 60 /* Saw any of the above 4 */
1511 
1512 /* These three values don't affect the parser. They are set but not used.
1513  */
1514 #define PNG_FP_WAS_VALID 64 /* Preceding substring is a valid fp number */
1515 #define PNG_FP_NEGATIVE 128 /* A negative number, including "-0" */
1516 #define PNG_FP_NONZERO 256 /* A non-zero value */
1517 #define PNG_FP_STICKY 448 /* The above three flags */
1518 
1519 /* This is available for the caller to store in 'state' if required. Do not
1520  * call the parser after setting it (the parser sometimes clears it.)
1521  */
1522 #define PNG_FP_INVALID 512 /* Available for callers as a distinct value */
1523 
1524 /* Result codes for the parser (boolean - true meants ok, false means
1525  * not ok yet.)
1526  */
1527 #define PNG_FP_MAYBE 0 /* The number may be valid in the future */
1528 #define PNG_FP_OK 1 /* The number is valid */
1529 
1530 /* Tests on the sticky non-zero and negative flags. To pass these checks
1531  * the state must also indicate that the whole number is valid - this is
1532  * achieved by testing PNG_FP_SAW_DIGIT (see the implementation for why this
