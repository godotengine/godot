412 struct MotionShapeSW : public ShapeSW {
413 
414  ShapeSW *shape;
415  Vector3 motion;
416 
417  virtual PhysicsServer::ShapeType get_type() const { return PhysicsServer::SHAPE_CONVEX_POLYGON; }
418 
419 
420  void project_range(const Vector3& p_normal, const Transform& p_transform, real_t &r_min, real_t &r_max) const {
421 
422  Vector3 cast = p_transform.basis.xform(motion);
423  real_t mina,maxa;
424  real_t minb,maxb;
425  Transform ofsb = p_transform;
426  ofsb.origin+=cast;
427  shape->project_range(p_normal,p_transform,mina,maxa);
428  shape->project_range(p_normal,ofsb,minb,maxb);
429  r_min=MIN(mina,minb);
430  r_max=MAX(maxa,maxb);
431  }
432 
433  Vector3 get_support(const Vector3& p_normal) const {
434 
435  Vector3 support = shape->get_support(p_normal);
436  if (p_normal.dot(motion)>0) {
437  support+=motion;
438  }
439  return support;
440  }
441  virtual void get_supports(const Vector3& p_normal,int p_max,Vector3 *r_supports,int & r_amount) const { r_amount=0; }
442  bool intersect_segment(const Vector3& p_begin,const Vector3& p_end,Vector3 &r_result, Vector3 &r_normal) const { return false; }
443 
444  Vector3 get_moment_of_inertia(float p_mass) const { return Vector3(); }
445 
446  virtual void set_data(const Variant& p_data) {}
447  virtual Variant get_data() const { return Variant(); }
