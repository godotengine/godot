118  float denom = (d00 * d11 - d01 * d01);
119  if (denom==0)
120  return uvs[0];
121  float v = (d11 * d20 - d01 * d21) / denom;
122  float w = (d00 * d21 - d01 * d20) / denom;
123  float u = 1.0f - v - w;
