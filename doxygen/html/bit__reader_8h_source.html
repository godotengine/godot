75 static WEBP_INLINE void VP8LoadNewBytes(VP8BitReader* const br) {
76  assert(br && br->buf_);
77  // Read 'BITS' bits at a time if possible.
78  if (br->buf_ + sizeof(lbit_t) <= br->buf_end_) {
79  // convert memory type to register type (with some zero'ing!)
80  bit_t bits;
81  lbit_t in_bits = *(lbit_t*)br->buf_;
82  br->buf_ += (BITS) >> 3;
83 #if !defined(__BIG_ENDIAN__)
84 #if (BITS == 32)
85 #if defined(__i386__) || defined(__x86_64__)
86  __asm__ volatile("bswap %k0" : "=r"(in_bits) : "0"(in_bits));
87  bits = (bit_t)in_bits; // 32b -> 64b zero-extension
88 #elif defined(_MSC_VER)
89  bits = _byteswap_ulong(in_bits);
90 #else
91  bits = (bit_t)(in_bits >> 24) | ((in_bits >> 8) & 0xff00)
92  | ((in_bits << 8) & 0xff0000) | (in_bits << 24);
93 #endif // x86
94 #elif (BITS == 16)
95  // gcc will recognize a 'rorw $8, ...' here:
96  bits = (bit_t)(in_bits >> 8) | ((in_bits & 0xff) << 8);
97 #endif
98 #else // LITTLE_ENDIAN
99  bits = (bit_t)in_bits;
100 #endif
101  br->value_ |= bits << br->missing_;
102  br->missing_ -= (BITS);
103  } else {
104  VP8LoadFinalBytes(br); // no need to be inlined
105  }
106 }

108 static WEBP_INLINE int VP8BitUpdate(VP8BitReader* const br, bit_t split) {
109  const bit_t value_split = split | (MASK);
110  if (br->missing_ > 0) { // Make sure we have a least BITS bits in 'value_'
111  VP8LoadNewBytes(br);
112  }
113  if (br->value_ > value_split) {
114  br->range_ -= value_split + 1;
115  br->value_ -= value_split + 1;
116  return 1;
117  } else {
118  br->range_ = value_split;
119  return 0;
120  }
121 }

123 static WEBP_INLINE void VP8Shift(VP8BitReader* const br) {
124  // range_ is in [0..127] interval here.
125  const int idx = br->range_ >> (BITS);
126  const int shift = kVP8Log2Range[idx];
127  br->range_ = kVP8NewRange[idx];
128  br->value_ <<= shift;
129  br->missing_ += shift;
130 }

132 static WEBP_INLINE int VP8GetBit(VP8BitReader* const br, int prob) {
133  // It's important to avoid generating a 64bit x 64bit multiply here.
134  // We just need an 8b x 8b after all.
135  const bit_t split =
136  (bit_t)((uint32_t)(br->range_ >> (BITS)) * prob) << ((BITS) - 8);
137  const int bit = VP8BitUpdate(br, split);
138  if (br->range_ <= (((bit_t)0x7e << (BITS)) | (MASK))) {
139  VP8Shift(br);
140  }
141  return bit;
142 }

144 static WEBP_INLINE int VP8GetSigned(VP8BitReader* const br, int v) {
145  const bit_t split = (br->range_ >> 1);
146  const int bit = VP8BitUpdate(br, split);
